// Oturum kontrolü document.addEventListener('DOMContentLoaded', function() { const currentUser = JSON.parse(localStorage.getItem('currentUser')); if (!currentUser) { alert('Bu sayfaya erişmek için giriş yapmalısınız!'); window.location.href = 'index.html'; return; } // Sayfa yenilendiğinde oturumu sonlandır window.addEventListener('beforeunload', function() { localStorage.removeItem('currentUser'); window.location.href = 'index.html'; }); // Müzik kontrolünü göster });
